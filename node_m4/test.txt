//READ ALL EQUIPOS BIEN

    curl -X GET http://localhost:3000/api/equipos

//READ UN ITEM DE EQUIPO POR ID BIEN

    curl -X GET http://localhost:3000/api/equipos/1

//DELETE EQUIPOS BIEN

    curl -X DELETE http://localhost:3000/api/equipos/1

//INSERT EQUIPO BIEN

curl -X POST http://localhost:3000/api/equipos -H "Content-Type: application/json" -d '{"id": 6, "idequipo":"insert", "equiponombre":"insert", "nacionequipo":"insert", "nombrepiloto":"insert", "nacionespilotos":"insert"}'

//UPDATE EQUIPOS BIEN
